Ordering

Controller 
$scope.items = [
    {city: 'New York', country: 'USA'},
    {city: 'Tokyo', country: 'Japan'},
    {city: 'Brisbane', country: 'Australia'}
];
Template: 
<li ng-repeat="place in items | orderBy: 'city'">
    <strong>{{ place.city }}</strong> / {{ place.country }}
</li>
Result:
  • {{ place.city }} / {{ place.country }}
